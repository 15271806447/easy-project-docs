# domresize 元素宽高变化

HTML元素（DOM）宽高变化后的监听事件

# 按需引入

import Domresize from 'easy-project-vue/dist/directives/domresize.js'

Vue.use(Domresize);

# 何时使用

echarts、baidu地图、等等canvas绘图插件，他们的初始化是一般是选定一个容器，比如echarts的初始化：

 <div id="id"></div>
echarts.init(document.getElementById(id))

这时候就带来一个问题，它的宽高一般是初始化时设定好的。

当窗口变化等情况让这个盒子的宽高变小或变大，这时候就得手动做宽高变更，因为这类插件一般用的canvas绘图。

所以我们需要一个钩子函数来处理这个事件，绑定这个钩子函数就是v-domresize指令。

# 用法示例

在使用v-domresize指令绑定DOM宽高变化后的监听事件，下面以echarts封装为例：


<template>
  <div :id="id" :style="{ width: width, height: height }" v-domresize="resizeCharts"></div>
</template>

<script>
import {v4 as uuidv4} from 'uuid'
import * as echarts from 'echarts'

export default {
  name: 'EChart',
  props: {
    title: {
      type: String,
      default: ''
    },
    height: {
      type: String,
      default: '250px'
    },
    width: {
      type: String,
      default: '100%'
    }
  },
  data() {
    return {
      id: '',
      myChart: null,
      option: {
        tooltip: {
          trigger: 'item'
        },
        legend: {
          top: '5%',
          left: 'center'
        },
        series: [
          {
            name: this.title,
            type: 'pie',
            radius: ['50%', '90%'],
            avoidLabelOverlap: false,
            label: {
              show: false,
              position: 'center',
              lineHeight: 16,
              fontSize: 16
            },
            emphasis: {
              label: {
                show: true,
                fontSize: '16',
                fontWeight: 'bold'
              }
            },
            labelLine: {
              show: false
            },
            data: []
          }
        ]
      }
    }
  },
  created() {
    // 利用uuid生成动态id
    this.id = uuidv4().replaceAll('-', '')
  },
  mounted() {
    this.initData()
  },
  beforeDestroy() {
    this.destroy()
  },
  methods: {
    resizeCharts() {
      // 当宽高变化时，调用echarts的resize方法来适配宽高。
      if (!this.myChart) {
        return
      }
      this.myChart.resize()
    },
    destroy() {
      this.myChart && this.myChart.dispose()
      this.myChart = null
    },
    initData() {
      this.destroy()
      // 初始化init
      this.myChart = echarts.init(document.getElementById(this.id))
      this.updateOption()
    },
    updateOption() {
      if (this.myChart) {
        this.myChart.clear()
        this.myChart.setOption(this.option, true)
      }
    }
  }
}
</script>
